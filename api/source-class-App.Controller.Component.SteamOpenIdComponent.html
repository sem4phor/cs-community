20:  * SteamOpenId Component
21:  *
22:  * Authorizes a user with help of the openid protocol and Steam as the openid provider.
23:  *
24:  * @author Valentin Rapp (Source of validate and genURL method: https://gist.github.com/Leimi/6904467)
25:  */
26: class SteamOpenIdComponent extends Component
27: {
28:     // the identifier of the openid provider
29:     const STEAM_LOGIN = 'https://steamcommunity.com/openid/login';
30: 
31:     /**
32:      * validate
33:      *
34:      * Validates an openid response of the openid provider and resends an authentication request
35:      *
36:      * @return string Returns the SteamID if successful or empty string on failure
37:      */
38:     public static function validate()
39:     {
40:         // Get the response params and build them properly
41:         $params = array(
42:             'openid.assoc_handle' => $_GET['openid_assoc_handle'],
43:             'openid.signed' => $_GET['openid_signed'],
44:             'openid.sig' => $_GET['openid_sig'],
45:             'openid.ns' => 'http://specs.openid.net/auth/2.0',
46:         );
47:         $signed = explode(',', $_GET['openid_signed']);
48:         foreach ($signed as $item) {
49:             $val = $_GET['openid_' . str_replace('.', '_', $item)];
50:             $params['openid.' . $item] = get_magic_quotes_gpc() ? stripslashes($val) : $val;
51:         }
52:         // Add the mode for checking authntication.
53:         $params['openid.mode'] = 'check_authentication';
54:         $data = http_build_query($params);
55:         // send data for validation
56:         $context = stream_context_create(array(
57:             'http' => array(
58:                 'method' => 'POST',
59:                 'header' =>
60:                     "Accept-language: en\r\n" .
61:                     "Content-type: application/x-www-form-urlencoded\r\n" .
62:                     "Content-Length: " . strlen($data) . "\r\n",
63:                 'content' => $data,
64:             ),
65:         ));
66:         // get the result form the stream context
67:         $result = file_get_contents(self::STEAM_LOGIN, false, $context);
68:         // Validate wheather it's true and if we have a good ID
69:         preg_match("#^http://steamcommunity.com/openid/id/([0-9]{17,25})#", $_GET['openid_claimed_id'], $matches);
70:         $steamID64 = is_numeric($matches[1]) ? $matches[1] : 0;
71:         return preg_match("#is_valid\s*:\s*true#i", $result) == 1 ? $steamID64 : false;
72:     }
73: 
74:     /**
75:      * Get the URL to sign into steam
76:      *
77:      * @param mixed returnTo URI to tell steam where to return
78:      * @param bool useAmp Use &amp; in the URL, true; or just &, false.
79:      * @return string The openid URL
80:      */
81:     public static function genUrl($returnTo = false, $useAmp = true)
82:     {
83:         $returnTo = (!$returnTo) ? (!empty($_SERVER['HTTPS']) ? 'https' : 'http') . '://' . $_SERVER['HTTP_HOST'] : $returnTo;
84:         $params = array(
85:             'openid.ns' => 'http://specs.openid.net/auth/2.0',
86:             'openid.mode' => 'checkid_setup',
87:             'openid.return_to' => $returnTo,
88:             'openid.realm' => (!empty($_SERVER['HTTPS']) ? 'https' : 'http') . '://' . $_SERVER['HTTP_HOST'],
89:             'openid.identity' => 'http://specs.openid.net/auth/2.0/identifier_select',
90:             'openid.claimed_id' => 'http://specs.openid.net/auth/2.0/identifier_select',
91:         );
92:         $sep = ($useAmp) ? '&amp;' : '&';
93:         return self::STEAM_LOGIN . '?' . http_build_query($params, '', $sep);
94:     }
95: }
