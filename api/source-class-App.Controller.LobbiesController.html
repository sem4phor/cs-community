 45:     /**
 46:      * isAuthorized
 47:      *
 48:      * Check if the provided user is authorized for the request:
 49:      * Banned users arent authorized for any action.
 50:      * All registered users can create one lobby.
 51:      * Only lobby-owners can kick users from their lobby.
 52:      * You can only delete your own lobby.
 53:      * You cant leave your own lobby.
 54:      * You can join a lobby if you arent already part of a lobby and have the required playtime.
 55:      *
 56:      *
 57:      * @param array|null $user The user to check the authorization of. If empty the user in the session will be used.
 58:      * @access public
 59:      * @return boolean True if $user is authorized, otherwise false
 60:      */
 61:     public function isAuthorized($user)
 62:     {
 63:         if (isset($user['role_id']) && $user['role_id'] === 4) {
 64:             return false;
 65:         }
 66:         // All registered users can add lobbies
 67:         if ($this->request->action === 'create' || $this->request->action === 'index') {
 68:             return true;
 69:         }
 70:         // kick only users in own lobby and not self
 71:         if ($this->request->action === 'kick') {
 72:             $steam_id = $this->request->params['pass'][0];
 73:             $lobby = $this->Lobbies->find()->where(['owner_id =' => $user['steam_id']])->contain(['Users'])->toArray()[0];
 74:             foreach ($lobby->users as $lobby_user) {
 75:                 if ($steam_id == $user['steam_id']) {
 76:                     return false;
 77:                 } elseif ($steam_id == $lobby_user->steam_id) {
 78:                     return true;
 79:                 }
 80:             }
 81:         }
 82:         // you can only delete your own lobby
 83:         if ($this->request->action == 'delete') {
 84:             $lobby_id = (int)$this->request->params['pass'][0];
 85:             if ($this->Lobbies->isOwnedBy($lobby_id, $user['steam_id'])) {
 86:                 return true;
 87:             }
 88:         }
 89:         // you can only leave if you are in the lobby but not own
 90:         if ($this->request->action == 'leave') {
 91:             $lobby_id = (int)$this->request->params['pass'][0];
 92:             $user_lobby_id = $this->Lobbies->Users->get($user['steam_id'])->lobby_id;
 93:             if ($this->Lobbies->isOwnedBy($lobby_id, $user['steam_id'])) {
 94:                 return false;
 95:             } elseif ($lobby_id === $user_lobby_id) {
 96:                 return true;
 97:             } else {
 98:                 return false;
 99:             }
100:         }
101:         // you are allowed to join if you arent already part of a lobby
102:         if ($this->request->action == 'join') {
103:             if ($this->Lobbies->Users->get($user['steam_id'])->lobby_id == null) {
104:                 return true;
105:             } else {
106:                 return false;
107:             }
108:         }
109:         return parent::isAuthorized($user);
110:     }

263:         if ($this->Lobbies->Users->save($user)) {

265:             if ($this->Lobbies->save($lobby)) {
266:                 $this->request->data['topic'] = 'lobby_join';
267:                 $this->request->data['lobby'] = $lobby;

269:                 $this->websocketSend($this->request->data);
270:                 if ($lobby->free_slots == 0) {
271:                     $this->request->data['topic'] = 'lobby_full';
